\addpersonalvariant{Михаил Бурмистров}


\tasknumber{1}\task{
    Сколько фотонов испускает за 20 минут лазер,
    если мощность его излучения $15{,}0\,\text{мВт}$.
    Длина волны излучения $600\,\text{нм}$.
}
\answer{
    $
        N = \frac{Pt\lambda}{hc}
           = \frac{
                15{,}0\,\text{мВт} \cdot 20 \cdot 60 \units{с} \cdot 600\,\text{нм}
            }{
                6{,}626 \cdot 10^{-34}\,\text{Дж}\cdot\text{с} \cdot 3 \cdot 10^{8}\,\frac{\text{м}}{\text{с}}
            }
           \approx {0{,}54}\cdot10^{20}\units{фотонов}
    $
}
\vspace{120pt}

\tasknumber{2}\task{
    В ядре электрически нейтрального атома 108 частиц.
    Вокруг ядра обращается 47 электронов.
    Сколько в ядре этого атома протонов и нейтронов?
}
\answer{
    $Z = 47$ протонов и $A - Z = 61$ нейтронов
}
\vspace{120pt}

}

\tasknumber{4}\task{
    Какой период полураспада радиоактивного изотопа,
    если за 24 ч в среднем распадается 75000 атомов из 80000?
}
\answer{
    $
        N(t) = N_0\cdot 2^{-\frac t{\tau_\frac12}}
        \implies \log_2\frac N{N_0} = - \frac t{\tau_\frac 12}
        \implies \tau_\frac 12 = - \frac t {\log_2\frac N{N_0}}
                                  =   \frac t {\log_2\frac {N_0}N}
        = \frac{
            24 \units{ч}
        }
        {
            \log_2\frac{80000}{80000 - 75000}
        }
        \approx 6{,}0\units{ч}
    $
}
