\addpersonalvariant{Михаил Бурмистров}


\tasknumber{1}\task{
        Запишите определения:
        \begin{itemize}
            \item гармонические колебания,
    \item   вынужденные колебания,\item   затухающие колебания.
        \end{itemize}
}
\vspace{120pt}

\tasknumber{2}\task{
    Определите частоту колебаний, если их период составляет $T = 4\,\text{мс}$.
}
\answer{
    $\nu = \frac 1T = \frac 1{ 4\,\text{мс} } = 250\,\text{Гц}$
}
\vspace{120pt}

\tasknumber{3}\task{
    Определите период колебаний, если их частота составляет $\nu = 4\,\text{кГц}$.
    Сколько колебаний произойдёт за $t = 5\,\text{мин}$?
}
\answer{
    \begin{align*}
        T &= \frac 1\nu = \frac 1{ 4\,\text{кГц} } = 0{,}25\,\text{мc}, \\
        N &= \nu t = 4\,\text{кГц}\cdot5\,\text{мин} = 1200000\,\text{колебаний}.
    \end{align*}
}
\vspace{120pt}

\tasknumber{4}\task{
    Амплитуда колебаний точки составляет $A = 2\,\text{см}$, а частота~--- $\nu = 5\,\text{Гц}$.
    Определите, какой путь преодолеет эта точка за $t = 10\,\text{с}$.
}
\answer{
    $
                s
                    = 4A \cdot N = 4A \cdot \frac tT = 4A \cdot t\nu
                    = 4 \cdot 2\,\text{см} \cdot 10\,\text{с} \cdot 5\,\text{Гц}
                    = 4{,}0\,\text{м}
            $
}
\vspace{120pt}

\tasknumber{5}\task{

    $A = 30\,\text{см}$, а период $T = 8\,\text{с}$.
}
