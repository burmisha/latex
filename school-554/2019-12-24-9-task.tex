\tasknumber{2}\task{
    Определите расстояние между вторым и десятым гребнями волн,
    если длина волны равна $5\,\text{м}$.
    Сколько между ними ещё уместилось гребней?
}
\answer{
    $
                l
                    = (n_2 - n_1) \cdot \lambda
                    = \cbr{10 - 2} \cdot 5\,\text{м}
                    = 40\,\text{м},
                \quad
                n = n_2 - n_1 - 1 = 10 - 2 - 1 = 7
            $
}

\tasknumber{3}\task{
    Определите скорость звука в среде, если источник звука,
    колеблющийся с периодом $6\,\text{мc}$, возбуждает волны длиной
    $2{,}1\,\text{м}$.
}
\answer{
    $
                \lambda
                    = vT \implies v
                    = \frac{\lambda}{T}
                    = \frac{ 2{,}1\,\text{м} }{ 6\,\text{мc} }
                    = { 350{,}0\,\frac{\text{м}}{\text{c}} }
            $
}

\tasknumber{4}\task{
    Мимо неподвижного наблюдателя прошло $5\,$ гребней волн за $10\,\text{c}$,
    начиная с первого.
    Каковы длина, период и частота волны,
    если скорость распространения волн $4\,\frac{\text{м}}{\text{с}}$?
}
\answer{
    \begin{align*}
        \lambda &= \frac L{N-1} = \frac {vt}{N-1} = \frac {4\,\frac{\text{м}}{\text{с}}\cdot10\,\text{c}}{5\, - 1} = 10{,}0\,\text{м}, \\
        T &= \frac {\lambda}{v} = \frac {vt}{\cbr{N-1}v} = \frac {t}{N-1} =  \frac { 10\,\text{c} }{5\, - 1} = 2{,}5\,\text{с}, \\
        \nu &= \frac 1T = \frac {N-1}{t} = \frac {5\, - 1}{ 10\,\text{c} } = 0{,}4\,\text{Гц}.
        \\
        &\text{Если же считать гребни целиком, т.е.
        не вычитать единицу:} \\
        \lambda' &= \frac L{N} = \frac {vt}{N} = \frac {4\,\frac{\text{м}}{\text{с}}\cdot10\,\text{c}}{ 5\, } = 8{,}0\,\text{м}, \\
        T' &= \frac {\lambda'}{v} = \frac {vt}{Nv} = \frac tN =  \frac { 10\,\text{c} }{ 5\, } = 2{,}0\,\text{с}, \\
        \nu' &= \frac 1{T'} = \frac {N}{t} = \frac { 5\, }{ 10\,\text{c} } = 0{,}5\,\text{Гц}.
    \end{align*}
}
