\tasknumber{1}\task{
    На резистор сопротивлением $R = 5\,\text{Ом}$ подали напряжение $V = 150\,\text{В}$.
    Определите ток, который потечёт через резистор, и мощность, выделяющуюся на нём.
}
\answer{
    \begin{align*}
    \mathcal{I} &= \frac{ V }{ R } = \frac{ 150\,\text{В} }{ 5\,\text{Ом} } = 30{,}0\,\text{А}, \\
    P &= \frac{ V^2 }{ R } = \frac{ \sqr{ 150\,\text{В} } }{ 5\,\text{Ом} } = 4500{,}0\,\text{Вт}
    \end{align*}
}

\tasknumber{2}\task{
    Замкнутая электрическая цепь состоит из ЭДС $\mathcal{E} = 20\,\text{В}$ и сопротивлением $r$
    и резистора $R = 30\,\text{Ом}$.
    Определите ток, протекающий в цепи.
    Какая тепловая энергия выделится на резисторе за время
    $\tau = 4\,\text{с}$? Какая работа будет совершена ЭДС за это время? Каков знак этой работы? Чему равен КПД цепи? Вычислите значения для 2 случаев:
    $r=0$ и $r = 4\,\text{Ом}$.
}
\answer{
    \begin{align*}
    \mathcal{I}_1
                    &= \frac{ \mathcal{E} }{ R }
                    = \frac{ 20\,\text{В} }{ 30\,\text{Ом} }
                    = 0{,}67\,\text{А}, \\
    \mathcal{I}_2
                    &= \frac{ \mathcal{E} }{ R + r }
                    = \frac{ 20\,\text{В} }{ 30\,\text{Ом} + 4\,\text{Ом} }
                    = 0{,}59\,\text{А}, \\
    Q_1
                    &= \mathcal{I}_1^2R\tau
                    = \cbr{ \frac{ \mathcal{E} }{ R } }^2 R \tau
                    = \cbr{ \frac{ 20\,\text{В} }{ 30\,\text{Ом} } }^2 \cdot 30\,\text{Ом} \cdot 4\,\text{с}
                    = 53{,}87\,\text{Дж}, \\
    Q_2
                    &= \mathcal{I}_2^2R\tau
                    = \cbr{\frac{ \mathcal{E} }{ R + r } }^2 R \tau
                    = \cbr{\frac{ 20\,\text{В} }{ 30\,\text{Ом} + 4\,\text{Ом} } }^2 \cdot 30\,\text{Ом} \cdot 4\,\text{с}
                    = 41{,}77\,\text{Дж}, \\
    A_1
                    &= \mathcal{I}_1\tau\mathcal{E}
                    = \frac{ \mathcal{E} }{ R } \tau \mathcal{E}
                    = \frac{ \mathcal{E}^2 \tau }{ R }
                    = \frac{ \sqr{ 20\,\text{В} } \cdot 4\,\text{с} }{ 30\,\text{Ом} }
                    = 53{,}6\,\text{Дж}, \text{положительна}, \\
    A_2
                    &= \mathcal{I}_2\tau\mathcal{E}
                    = \frac{ \mathcal{E} }{ R + r } \tau \mathcal{E}
                    = \frac{ \mathcal{E}^2 \tau }{ R + r }
                    = \frac{ \sqr{ 20\,\text{В} } \cdot 4\,\text{с} }{ 30\,\text{Ом} + 4\,\text{Ом} }
                    = 47{,}2\,\text{Дж}, \text{положительна}, \\
    \eta_1
                    &= \frac{ Q_1 }{ A_1 }
                    = \ldots
                    = \frac{ R }{ R }
                    = 1, \\
    \eta_2
                    &= \frac{ Q_2 }{ A_2 }
                    = \ldots
                    = \frac{ R }{ R + r }
                    = 0{,}88\,
    \end{align*}
}

\tasknumber{3}\task{
    Лампочки, сопротивления которых $R_1 = 1{,}0\,\text{Ом}$ и $R_2 = 4{,}0\,\text{Ом}$, поочерёдно подключённные к некоторому источнику тока,
    потребляют одинаковую мощность.
    Найти внутреннее сопротивление источника и КПД цепи в каждом случае.
}
\answer{
    \begin{align*}
    P_1 &= \sqr{ \frac{ \mathcal{E} }{ R_1 + r } } R_1,
    P_2  = \sqr{ \frac{ \mathcal{E} }{ R_2 + r } } R_2,
    P_1 = P_2 \implies \\
    &\implies R_1 \sqr{ R_2 + r } = R_2 \sqr{ R_1 + r } \implies \\
    &\implies R_1 R_2^2 + 2 R_1 R_2 r + R_1 r^2 =
                R_2 R_1^2 + 2 R_2 R_1 r + R_2 r^2  \implies \\
    &\implies r^2 (R_2 - R_1) = R_2^2 R_2 - R_1^2 R_2 \implies \\
    &\implies r
        = \sqrt{ R_1 R_2 \frac{ R_2 - R_1 }{ R_2 - R_1 } }
        = \sqrt{ R_1 R_2 }
        = \sqrt{ 1{,}0\,\text{Ом} \cdot 4{,}0\,\text{Ом} }
        = 2{,}0\,\text{Ом}.
        \\
     \eta_1
        &= \frac{ R_1 }{ R_1 + r }
        = \frac{ \sqrt{ R_1 } }{ \sqrt{ R_1 } + \sqrt{ R_2 } }
        = 0{,}333\,, \\
     \eta_2
        &= \frac{ R_2 }{ R_2 + r }
        = \frac{ \sqrt{ R_2 } }{ \sqrt{ R_2 } + \sqrt{ R_1 } }
        = 0{,}667\,
    \end{align*}
}
