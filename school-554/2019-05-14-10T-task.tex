\task{%
    На резистор сопротивлением $r = 18{,}0\,\text{Ом}$ подали напряжение $V = 120{,}0\,\text{В}$.
    Определите ток, который потечёт через резистор, и мощность, выделяющуюся на нём.
}
\answer{%
    \begin{align*}
    \mathcal{I} &= \frac{ V }{ r } = \frac{ 120{,}0\,\text{В} }{ 18{,}0\,\text{Ом} } = 6{,}67\,\text{А},  \\
    P &= \frac{ V^2 }{ r } = \frac{ \sqr{ 120{,}0\,\text{В} } }{ 18{,}0\,\text{Ом} } = 800{,}00\,\text{Вт}
    \end{align*}
}

\tasknumber{2}%
\task{%
    Замкнутая электрическая цепь состоит из ЭДС $\mathcal{E} = 1{,}0\,\text{В}$ и сопротивлением $r$
    и резистора $R = 30\,\text{Ом}$.
    Определите ток, протекающий в цепи.
    Какая тепловая энергия выделится на резисторе за время
    $\tau = 10{,}0\,\text{с}$? Какая работа будет совершена ЭДС за это время? Каков знак этой работы? Чему равен КПД цепи? Вычислите значения для 2 случаев:
    $r=0$ и $r = 10{,}0\,\text{Ом}$.
}
\answer{%
    \begin{align*}
    \mathcal{I}_1 &= \frac{ \mathcal{E} }{ R } = \frac{ 1{,}0\,\text{В} }{ 30\,\text{Ом} } = 0{,}03\,\text{А},  \\
    \mathcal{I}_2 &= \frac{ \mathcal{E} }{ R + r } = \frac{ 1{,}0\,\text{В} }{ 30\,\text{Ом} + 10{,}0\,\text{Ом} } = 0{,}03\,\text{А},  \\
    Q_1 &= \mathcal{I}_1^2R\tau = \sqr{ \frac{ \mathcal{E} }{ R } } R \tau
            = \sqr{ \frac{ 1{,}0\,\text{В} }{ 30\,\text{Ом} } } \cdot 30\,\text{Ом} \cdot \cdot 10{,}0\,\text{с} = 0{,}270\,\text{Дж},  \\
    Q_2 &= \mathcal{I}_2^2R\tau = \sqr{ \frac{ \mathcal{E} }{ R + r } } R \tau
            = \sqr{ \frac{ 1{,}0\,\text{В} }{ 30\,\text{Ом} + 10{,}0\,\text{Ом} } } \cdot 30\,\text{Ом} \cdot 10{,}0\,\text{с} = 0{,}270\,\text{Дж},  \\
    A_1 &= \mathcal{I}_1\tau\mathcal{E} = \frac{ \mathcal{E} }{ R } \tau \mathcal{E}
            = \frac{ \mathcal{E}^2 \tau }{ R } = \frac{ \sqr{ 1{,}0\,\text{В} } \cdot 10{,}0\,\text{с} }{ 30\,\text{Ом} }
            = 0{,}300\,\text{Дж}, \text{ положительна },  \\
    A_2 &= \mathcal{I}_2\tau\mathcal{E} = \frac{ \mathcal{E} }{ R + r } \tau \mathcal{E}
            = \frac{ \mathcal{E}^2 \tau }{ R + r } = \frac{ \sqr{ 1{,}0\,\text{В} } \cdot 10{,}0\,\text{с} }{ 30\,\text{Ом} + 10{,}0\,\text{Ом} }
            = 0{,}300\,\text{Дж}, \text{ положительна },  \\
    \eta_1 &= \frac{ Q_1 }{ A_1 } = \ldots = \frac{ R }{ R } = 1,  \\
    \eta_2 &= \frac{ Q_2 }{ A_2 } = \ldots = \frac{ R }{ R + r } = 0{,}90
    \end{align*}
}

\tasknumber{3}%
\task{%
    Лампочки, сопротивления которых $R_1 = 3{,}00\,\text{Ом}$ и $R_2 = 48{,}00\,\text{Ом}$, поочерёдно подключённные к некоторому источнику тока,
    потребляют одинаковую мощность.
    Найти внутреннее сопротивление источника и КПД цепи в каждом случае.
}
\answer{%
    \begin{align*}
    P_1 &= \sqr{ \frac{ \mathcal{E} }{ R_1 + r } }R_1,
        P_2  = \sqr{ \frac{ \mathcal{E} }{ R_2 + r } }R_2,
        P_1 = P_2 \implies  \\
        &\implies R_1 \sqr{ R_2 + r } = R_2 \sqr{ R_1 + r } \implies  \\
        &\implies R_1 R_2^2 + 2 R_1 R_2 r + R_1 r^2 =
                    R_2 R_1^2 + 2 R_2 R_1 r + R_2 r^2  \implies  \\
    &\imuplies r^2 (R_2 - R_1) = R_2^2 R_2 - R_1^2 R_2 \implies  \\
    &\implies r
            = \sqrt{ R_1 R_2 \frac{ R_2 - R_1 }{ R_2 - R_1 } }
            = \sqrt{ R_1 R_2 }
            = \sqrt{ 3{,}00\,\text{Ом} \cdot 48{,}00\,\text{Ом} }
            = 12{,}00\,\text{Ом}.
            \\
    \eta_1
            &= \frac{ R_1 }{ R_1 + r }
            = \frac{ \sqrt{ R_1 } }{ \sqrt{ R_1 } + \sqrt{ R_2 } }
            = 0{,}200,  \\
    \eta_2
            &= \frac{ R_2 }{ R_2 + r }
            = \frac{ \sqrt{ R_2 } }{ \sqrt{ R_2 } + \sqrt{ R_1 } }
            = 0{,}800
    \end{align*}
}
