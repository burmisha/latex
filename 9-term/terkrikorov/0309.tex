\begin{enumerate}
	\item частные производные $f_x, f_y$ удовлетворяют условию Липшица,

\end{enumerate}

Теорема Тихонова. Если выполнены 1--3, то $\foral{[0,T]} \exist{ C(t), \eps_0(T)} \text{ при } 0<\eps<\eps_0, |x_{10}|<\mu_0$ сушествует решение и $x_1(t) = x_{10}(t) + \obol{\eps}, x_2(t,\eps) = x_{20}(t) + y\cbr{\frac t\eps} + \obol{\eps}.$

Лемма 1. 
\al{
	&\modul{f_2(x_{10}(t), x_{20}(t) + y\cbr{\frac t\eps}, t)-f_2(0,\mu+y\cbr{\frac t\eps}, 0)} = \\ 
	&=\modul{\integr 01 \ddd{}u\cbr{f_2\cbr{x_{10}(t), x_{20}(t) + uy\cbr{\frac t\eps}, t}-f_2\cbr{0,\mu+uy\cbr{\frac t\eps}, 0}} du} = \\
	&= \modul{\integr 01 \cbr{\dd{f_2}{x_2}\cbr{x_{10}(t), x_{20}(t) + uy\cbr{\frac t\eps}, t} 
	 	-\dd{f_2}{x_2}\cbr{0,\mu+uy\cbr{\frac t\eps}, 0}} du} \le \\
	&\le C_1e^{-\frac{\delta t}\eps}\cbr{|x_1(t) - x_1(0)| + |x_2(t) - x_2(0)| + t} \le C_2te^{-\frac\delta\eps t}. 
}

Лемма 1.
\al{
	\modul{f_2(x_{10}(t),x_{20} + y\cbr{\frac t\eps}, t) - f_2(0, \mu +y\cbr{\frac t\eps}, 0)} \le C_2(t)te^{-\frac{\delta t}\eps}
}

Лемма 2.
\al{
	|H_2(s,t,\eps)| \le e^{-\frac \delta\eps(t-s)}, \qquad 0\le s\le t.
}
\al{

Хотим применить теорему Канторовича. Необходимо проверить
\begin{itemize}
	\item $F'(z)$ удовлетворяет условию Липшица по $z$. 
	Да: смотрим на свойства подынтегральных функций: 
	$|F_1'(z) - F_1'(\tilde z)| \le C(T)|z - \tilde z|$, 
	$|F_2'(z) - F_2'(\tilde z)| \le C(T)\norm{z - \tilde z}\integr 0t \frac1\eps H(s,t,\eps)ds,$ $\integr 0t \frac1\eps e^{-\frac\delta\eps(t-s)}ds = \skipped{2--3}$
	\item $I - F'(0)$ имеет ограниченный обратный. Т.е. разрешима система.

	}
